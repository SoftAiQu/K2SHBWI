class CompressionType(Enum):
    """Supported compression algorithms"""
    NONE = 0
    ZLIB = 1
    BROTLI = 2
    LZMA = 3
    ZSTD = 4  # Added support for Zstandard
    
    @classmethod
    def get_compressor(cls, comp_type: 'CompressionType') -> callable:
        """Get the compression function for the given type"""
        if comp_type == cls.NONE:
            return lambda x: x
        elif comp_type == cls.ZLIB:
            return lambda x: zlib.compress(x, level=9)
        elif comp_type == cls.BROTLI:
            import brotli
            return lambda x: brotli.compress(x)
        elif comp_type == cls.LZMA:
            import lzma
            return lambda x: lzma.compress(x)
        elif comp_type == cls.ZSTD:
            # Defer import of zstandard until actually used. If it's not
            # installed, raise a descriptive CompressionError.
            try:
                try:
                    # dynamic import to avoid top-level import errors in environments
                    # where zstandard isn't installed (helps linters/IDEs too)
                    import importlib
                    zstd = importlib.import_module('zstandard')
                except Exception:
                    raise CompressionError(
                        "zstandard (zstd) library is not installed. "
                        "Install with: pip install zstandard"
                    )
            except Exception as e:  # ImportError or similar
                raise CompressionError(
                    "zstandard (zstd) library is not installed. "
                    "Install with: pip install zstandard"
                )

            def zstd_compress(b: bytes) -> bytes:
                c = zstd.ZstdCompressor()
                return c.compress(b)

            return zstd_compress
        raise ValueError(f"Unsupported compression type: {comp_type}")

    @classmethod
    def get_decompressor(cls, comp_type: 'CompressionType') -> callable:
        """Get the decompression function for the given type"""
        if comp_type == cls.NONE:
            return lambda x: x
        elif comp_type == cls.ZLIB:
            return zlib.decompress
        elif comp_type == cls.BROTLI:
            import brotli
            return brotli.decompress
        elif comp_type == cls.LZMA:
            import lzma
            return lzma.decompress
        elif comp_type == cls.ZSTD:
            try:
                import zstandard as zstd
            except Exception:
                raise CompressionError(
                    "zstandard (zstd) library is not installed. "
                    "Install with: pip install zstandard"
                )

            def zstd_decompress(b: bytes) -> bytes:
                d = zstd.ZstdDecompressor()
                return d.decompress(b)

            return zstd_decompress
        raise ValueError(f"Unsupported compression type: {comp_type}")
